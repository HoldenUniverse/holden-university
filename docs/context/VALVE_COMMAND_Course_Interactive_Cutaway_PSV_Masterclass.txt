1. How to Simulate the Valve in Operation
Think of the valve as a set of moving internal parts.

For a PSV, the main operational states are:




Plain Text
Normal Operation:
Pressure below set pressure. Disc seated. No relief flow.
 
Approaching Set Pressure:
Pressure increases. Disc still closed.
 
Set Pressure:
Pressure reaches set point. Disc begins to lift.
 
Full Lift:
Valve opens. Flow discharges through outlet.
 
Blowdown / Reseat:
Pressure drops. Spring pushes disc back onto seat.
 
Failure:
Valve leaks, chatters, fouls, or fails to reseat.

Show more lines
In code, you would control this with a pressure slider.

Example logic:




TypeScript
const setPressure = 150;
const fullLiftPressure = 175;
 
function calculateValveLift(pressure: number) {
  if (pressure < setPressure) return 0;
 
  if (pressure >= setPressure && pressure < fullLiftPressure) {
    return (pressure - setPressure) / (fullLiftPressure - setPressure);
  }
 
  return 1;
}
Then pass that value into the valve model:




TypeScript
const valveLift = calculateValveLift(pressure);
 
<PSVModel
  pressure={pressure}
  valveLift={valveLift}

  showFlow={showFlow}
/>

function getOperatingState(pressure: number, setPressure: number) {
  if (pressure < setPressure * 0.9) {

function FoulingDeposits({ severity }) {

import { useFrame } from "@react-three/fiber";
import { useRef } from "react";
 
function Disc({ targetLift }) {
  const discRef = useRef();
 
  useFrame(() => {
    if (discRef.current) {
      const currentY = discRef.current.position.y;
      const targetY = targetLift * 0.65;
 
      discRef.current.position.y += (targetY - currentY) * 0.08;

7. How to Simulate Chatter
Chatter is when the disc rapidly opens/closes due to unstable flow, excessive backpressure, improper sizing, or system instability.

In the simulator, add a failure mode:




Plain Text
Failure Mode: Chatter
Then animate the disc rapidly:




TypeScript
function calculateChatterLift(baseLift, failureMode, time) {
  if (failureMode !== "chatter") return baseLift;
 
  return baseLift + Math.sin(time * 35) * 0.12;
}
Inside useFrame:




TypeScript
useFrame((state) => {

import React, { useMemo, useRef, useState } from "react";
import { Canvas, useFrame } from "@react-three/fiber";
import { OrbitControls, Html } from "@react-three/drei";
import * as THREE from "three";
 
type FailureMode =
  | "clean"
  | "fouled"
  | "leaking-seat"
  | "damaged-seat"
  | "chatter"
  | "stuck-open"
  | "failed-to-lift";
 
const SET_PRESSURE = 150;
const FULL_LIFT_PRESSURE = 175;
 
function calculateValveLift(pressure: number, failureMode: FailureMode) {
  if (failureMode === "stuck-open") return 1;
  if (failureMode === "failed-to-lift") return 0;
 
  if (pressure < SET_PRESSURE) return 0;
 
  if (pressure >= SET_PRESSURE && pressure < FULL_LIFT_PRESSURE) {
    return (pressure - SET_PRESSURE) / (FULL_LIFT_PRESSURE - SET_PRESSURE);
  }
 
  return 1;
}
 
function getOperatingState(pressure: number, failureMode: FailureMode) {
  if (failureMode === "stuck-open") return "Failure: PSV stuck open.";
  if (failureMode === "failed-to-lift") return "Failure: PSV failed to lift.";
  if (failureMode === "chatter" && pressure >= SET_PRESSURE) {
    return "Failure: PSV chatter detected.";
  }
  if (pressure < SET_PRESSURE * 0.9) return "Normal operation: PSV seated.";
  if (pressure < SET_PRESSURE) return "Approaching set pressure.";
  if (pressure < FULL_LIFT_PRESSURE) return "PSV opening: disc lifting.";
  return "Full lift: relief flow active.";
}

function FoulingDeposits({ active }: { active: boolean }) {
  const deposits = useMemo(() => {
    return Array.from({ length: 28 }).map((_, index) => {
      const angle = (index / 28) * Math.PI * 2;
      const radius = 0.42 + Math.random() * 0.08;
      return {
        position: [
          Math.cos(angle) * radius,
          -0.25 + Math.random() * 0.25,
          Math.sin(angle) * radius,
        ] as [number, number, number],
        scale: [
          0.04 + Math.random() * 0.05,
          0.03 + Math.random() * 0.04,
          0.04 + Math.random() * 0.05,
        ] as [number, number, number],
      };
    });
  }, []);
 
  if (!active) return null;
 
  return (
    <group>
      {deposits.map((deposit, index) => (
        <mesh key={index} position={deposit.position} scale={deposit.scale}>
          <sphereGeometry args={[1, 8, 8]} />
          <meshStandardMaterial color="#5a3921" roughness={1} />
        </mesh>
      ))}
    </group>
  );
}

function PSVModel({
  pressure,
  failureMode,
  showLabels,
  showFlow,
  cutawayMode,
  onSelectPart,
}: {
  pressure: number;
  failureMode: FailureMode;
  showLabels: boolean;
  showFlow: boolean;
  cutawayMode: boolean;
  onSelectPart: (part: string) => void;
}) {
  const discRef = useRef<THREE.Mesh>(null);
  const springRef = useRef<THREE.Mesh>(null);
 
  const baseLift = calculateValveLift(pressure, failureMode);
 
  useFrame((state) => {
    if (!discRef.current) return;
 
    let targetY = baseLift * 0.65;
 
    if (failureMode === "chatter" && pressure >= SET_PRESSURE) {
      targetY += Math.sin(state.clock.getElapsedTime() * 35) * 0.08;
    }
 
    discRef.current.position.y += (targetY - discRef.current.position.y) * 0.12;
 
    if (springRef.current) {
      springRef.current.scale.y = 1 - baseLift * 0.25;
    }
  });
 
  const bodyOpacity = cutawayMode ? 0.38 : 1;
 
  return (
    <group>
      {/* Valve Body */}
      <mesh position={[0, 0, 0]}>
        <cylinderGeometry args={[0.75, 0.75, 1.45, 48]} />
        <meshStandardMaterial
          color="#5f6b73"
          transparent={cutawayMode}
          opacity={bodyOpacity}
          roughness={0.55}
          metalness={0.45}
        />
      </mesh>
 
      {/* Inlet Nozzle */}
      <mesh position={[0, -0.85, 0]}>
        <cylinderGeometry args={[0.35, 0.35, 0.65, 32]} />
        <meshStandardMaterial color="#777f86" metalness={0.45} roughness={0.5} />
      </mesh>
 
      {/* Outlet */}
      <mesh position={[0.85, 0.1, 0]} rotation={[0, 0, Math.PI / 2]}>
        <cylinderGeometry args={[0.28, 0.28, 1.1, 32]} />
        <meshStandardMaterial color="#777f86" metalness={0.45} roughness={0.5} />
      </mesh>
 
      {/* Bonnet */}
      <mesh position={[0, 1.05, 0]}>
        <cylinderGeometry args={[0.42, 0.42, 0.9, 32]} />
        <meshStandardMaterial color="#4d5860" metalness={0.45} roughness={0.5} />
      </mesh>
 
      {/* Spring */}
      <mesh ref={springRef} position={[0, 1.08, 0]}>
        <torusKnotGeometry args={[0.18, 0.025, 120, 8]} />
        <meshStandardMaterial color="#c9d1d9" metalness={0.8} roughness={0.25} />
      </mesh>
 
      {/* Disc */}
      <mesh ref={discRef} position={[0, 0, 0]}>
        <cylinderGeometry args={[0.43, 0.43, 0.12, 32]} />
        <meshStandardMaterial color="#d7dde3" metalness={0.8} roughness={0.22} />
      </mesh>
 
      {/* Seat */}
      <mesh position={[0, -0.18, 0]} rotation={[Math.PI / 2, 0, 0]}>
        <torusGeometry args={[0.43, 0.025, 16, 64]} />
        <meshStandardMaterial color="#c7b177" metalness={0.6} roughness={0.3} />
      </mesh>
 
      <FoulingDeposits active={failureMode === "fouled"} />
 
      <LeakStream
        active={
          failureMode === "leaking-seat" ||
          failureMode === "damaged-seat" ||
          failureMode === "fouled"
        }
      />
 
      <DamagedSeat active={failureMode === "damaged-seat"} />
 
      {showFlow && baseLift > 0 && (
        <>
          <FlowArrow position={[0, -0.8, 0]} rotation={[0, 0, 0]} />
          <FlowArrow position={[0.75, 0.1, 0]} rotation={[0, 0, -Math.PI / 2]} />
        </>
      )}
 
      {showLabels && (
        <>
          <ValveLabel
            position={[0, 1.65, 0]}
            title="Spring"
            onClick={() => onSelectPart("spring")}
          />
          <ValveLabel
            position={[0.55, 0.2, 0]}
            title="Disc"
            onClick={() => onSelectPart("disc")}
          />
          <ValveLabel
            position={[-0.65, -0.18, 0]}
            title="Seat"
            onClick={() => onSelectPart("seat")}
          />
          <ValveLabel
            position={[0, -1.25, 0]}
            title="Nozzle"
            onClick={() => onSelectPart("nozzle")}
          />
          <ValveLabel
            position={[0.65, 1.05, 0]}
            title="Bonnet"
            onClick={() => onSelectPart("bonnet")}
          />
          <ValveLabel
            position={[1.45, 0.1, 0]}
            title="Outlet"
            onClick={() => onSelectPart("outlet")}
          />
        </>
      )}
    </group>
  );
}
